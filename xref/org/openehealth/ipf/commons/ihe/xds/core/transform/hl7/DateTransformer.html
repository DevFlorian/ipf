16  package org.openehealth.ipf.commons.ihe.xds.core.transform.hl7;
17  
18  import ca.uhn.hl7v2.model.DataTypeException;
19  import ca.uhn.hl7v2.model.primitive.CommonTS;
20  import org.joda.time.DateTime;
21  import org.joda.time.DateTimeZone;
22  import org.joda.time.format.DateTimeFormat;
23  import org.joda.time.format.DateTimeFormatter;
24  import org.openehealth.ipf.commons.ihe.xds.core.validate.ValidationMessage;
25  import org.openehealth.ipf.commons.ihe.xds.core.validate.XDSMetaDataException;
26  
27  /**
28   * Transfers between HL7 DTM timestamps used in IHE XDS profiles and {@link DateTime}.
29   */
30  public class DateTransformer {
31      private static final DateTimeFormatter FORMATTER = DateTimeFormat.forPattern("yyyyMMddHHmmss");
32  
33      /**
34       * Converts from the HL7 DTM format to {@link DateTime}.
35       *
36       * @param timestamp
37       *      String of the pattern <code>YYYY[MM[DD[HHMM[SS[.S[S[S[S]]]]]]]][+/-ZZZZ]</code>.
38       * @return
39       *      {@link DateTime} object, or <code>null</code> is the parameter was <code>null</code>.
40       */
41      public static DateTime fromHL7(String timestamp) {
42          if (timestamp == null) {
43              return null;
44          }
45  
46          // default time zone is UTC
47          if ((timestamp.indexOf('-') < 0) && (timestamp.indexOf('+') < 0)) {
48              timestamp += "+0000";
49          }
50  
51          try {
52              CommonTS ts = new CommonTS(timestamp);
53              DateTime dateTime = new DateTime(
54                      ts.getYear(),
55                      (ts.getMonth() == 0) ? 1 : ts.getMonth(),
56                      (ts.getDay() == 0) ? 1 : ts.getDay(),
57                      ts.getHour(),
58                      ts.getMinute(),
59                      ts.getSecond(),
60                      DateTimeZone.forOffsetHoursMinutes(
61                              ts.getGMTOffset() / 100,
62                              ts.getGMTOffset() % 100));
63              return dateTime.toDateTime(DateTimeZone.UTC);
64          } catch (DataTypeException e) {
65              throw new XDSMetaDataException(ValidationMessage.INVALID_TIME, timestamp);
66          }
67      }
68  
69      /**
70       * Converts from {@link DateTime} to the HL7 DTM format.
71       * According to IHE requirements, time zone is UTC, and milliseconds are not used.
72       *
73       * @param dateTime
74       *      {@link DateTime} object.
75       * @return
76       *      String of the pattern <code>YYYYMMDDhhmmss</code>.
77       */
78      public static String toHL7(DateTime dateTime) {
79          if (dateTime == null) {
80              return null;
81          }
82  
83          return FORMATTER.print(dateTime.toDateTime(DateTimeZone.UTC));
84      }
85  }
