112     /**
113      * Audits an intra-community (ITI-57) or cross-community (ITI_X1) Update Document Set events.
114      *
115      * @param serverSide            <code>true</code> for the Document Administrator / XCMU Initiating Gateway actor,
116      *                              <code>false</code> for the Document Registry / XCMU Responding Gateway actor.
117      * @param eventTypeCode         transaction code (ITI-57 or CH-XCMU)
118      * @param eventOutcome          event outcome indicator
119      * @param sourceUserId          ID of the user which triggered the transaction
120      * @param registryEndpointUri   Web service endpoint URI for the document registry
121      * @param submissionSetUniqueId UniqueID of the Submission Set registered
122      * @param homeCommunityId       home community ID (optional for ITI-57, required for CH-XCMU).
123      * @param patientId             Patient Id that this submission pertains to
124      * @param purposesOfUse         &lt;PurposeOfUse&gt; attributes from XUA SAML assertion.
125      * @param userRoles             user role codes from XUA SAML assertion.
126      */
127     private void auditUpdateDocumentSet(
128             boolean serverSide,
129             IHETransactionEventTypeCodes eventTypeCode,
130             RFC3881EventCodes.RFC3881EventOutcomeCodes eventOutcome,
131             String sourceUserId,
132             String sourceIpAddress,
133             String humanUserName,
134             String registryEndpointUri,
135             String submissionSetUniqueId,
136             String homeCommunityId,
137             String patientId,
138             List<CodedValueType> purposesOfUse,
139             List<CodedValueType> userRoles)
140     {
141         if (! isAuditorEnabled()) {
142             return;
143         }
144 
145         GenericIHEAuditEventMessage event = new GenericIHEAuditEventMessage(
146                 ! serverSide,
147                 eventOutcome,
148                 RFC3881EventCodes.RFC3881EventActionCodes.UPDATE,
149                 serverSide ? new DICOMEventIdCodes.Import() : new DICOMEventIdCodes.Export(),
150                 eventTypeCode,
151                 purposesOfUse);
152 
153         event.setAuditSourceId(getAuditSourceId(), getAuditEnterpriseSiteId());
154 
155         event.addSourceActiveParticipant(
156                 sourceUserId,
157                 serverSide ? null : getSystemAltUserId(),
158                 null,
159                 serverSide ? sourceIpAddress : getSystemNetworkId(),
160                 true);
161 
162         if (!EventUtils.isEmptyOrNull(humanUserName)) {
163             event.addHumanRequestorActiveParticipant(humanUserName, null, humanUserName, userRoles);
164         }
165 
166         event.addDestinationActiveParticipant(
167                 registryEndpointUri,
168                 serverSide ? getSystemAltUserId() : null,
169                 null,
170                 EventUtils.getAddressForUrl(registryEndpointUri, false),
171                 false);
172 
173         if (!EventUtils.isEmptyOrNull(patientId)) {
174             event.addPatientParticipantObject(patientId);
175         }
176 
177         List<TypeValuePairType> pairs = new ArrayList<>();
178         if (StringUtils.isNotEmpty(homeCommunityId)) {
179             TypeValuePairType pair = new TypeValuePairType();
180             pair.setType("ihe:homeCommunityID");
181             pair.setValue(homeCommunityId.getBytes(StandardCharsets.UTF_8));
182             pairs.add(pair);
183         }
184 
185         event.addParticipantObjectIdentification(
186                 new IHETransactionParticipantObjectIDTypeCodes.SubmissionSet(),
187                 null,
188                 null,
189                 pairs,
190                 submissionSetUniqueId,
191                 RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeCodes.SYSTEM,
192                 RFC3881ParticipantObjectCodes.RFC3881ParticipantObjectTypeRoleCodes.JOB,
193                 null,
194                 null);
195 
196         audit(event);
197     }

236      * @param clientIpAddress       IP address of the Document On-Demand Document Source actor.
237      * @param submissionSetUniqueId unique ID of the XDS submission set.
238      * @param patientId             patient ID as an HL7 v2 CX string.
239      * @param purposesOfUse         &lt;PurposeOfUse&gt; attributes from XUA SAML assertion.
240      * @param userRoles             user role codes from XUA SAML assertion.

242     public void auditIti61(
243             boolean serverSide,
244             RFC3881EventCodes.RFC3881EventOutcomeCodes eventOutcome,
245             String userId,
246             String userName,
247             String serviceEndpointUri,
248             String clientIpAddress,
249             String submissionSetUniqueId,

253     {

322 
323         event.addDestinationActiveParticipant(
324                 serviceEndpointUri,

327                 serverSide ? getSystemNetworkId() : EventUtils.getAddressForUrl(serviceEndpointUri, false),

329 
330         event.setAuditSourceId(getAuditSourceId(), getAuditEnterpriseSiteId());

332         if (!EventUtils.isEmptyOrNull(patientId)) {

591     private void doAuditImagingEvent(
592             boolean serverSide,
593             IHETransactionEventTypeCodes transactionEventTypeCodes,
594             RFC3881EventCodes.RFC3881EventOutcomeCodes eventOutcome,
595             String userId,
596             String userName,
597             String serviceEndpointUri,
598             String clientIpAddress,
599             String[] studyInstanceUniqueIds,
600             String[] seriesInstanceUniqueIds,
601             String[] documentUniqueIds,
602             String[] repositoryUniqueIds,
603             String[] homeCommunityIds,
604             String patientId,
605             List<CodedValueType> purposesOfUse,
606             List<CodedValueType> userRoles)
607     {
608         ImagingRetrieveEvent event = new ImagingRetrieveEvent(
609                 ! serverSide,
610                 eventOutcome,
611                 transactionEventTypeCodes,
612                 purposesOfUse);
613 
614         event.addSourceActiveParticipant(
615                 serviceEndpointUri,
616                 serverSide ? getSystemAltUserId() : null,
617                 null,
618                 serverSide ? getSystemNetworkId() : EventUtils.getAddressForUrl(serviceEndpointUri, false),
619                 false);
620 
621         event.addDestinationActiveParticipant(
622                 userId,
623                 serverSide ? null : getSystemAltUserId(),
624                 null,
625                 serverSide ? clientIpAddress : getSystemNetworkId(),
626                 true);
627 
628         if (! EventUtils.isEmptyOrNull(userName)) {
629             event.addHumanRequestorActiveParticipant(userName, null, userName, userRoles);
630         }
631 
632         event.setAuditSourceId(getAuditSourceId(), getAuditEnterpriseSiteId());
633 
634         if (! EventUtils.isEmptyOrNull(patientId)) {
635             event.addPatientParticipantObject(patientId);
636         }
637 
638         if (! EventUtils.isEmptyOrNull(documentUniqueIds)) {
639             for (int i = 0; i < documentUniqueIds.length; i++) {
640                 event.addDocumentParticipantObject(
641                         studyInstanceUniqueIds[i],
642                         seriesInstanceUniqueIds[i],
643                         documentUniqueIds[i],
644                         repositoryUniqueIds[i],
645                         homeCommunityIds[i]);
646             }
647         }
648 
649         audit(event);
650     }
