52      public static ObjectRole fromCode(String codeString) throws FHIRException {
53          if (codeString == null || "".equals(codeString))
54              return null;
55  
56          for (ObjectRole objectRole : ObjectRole.values()) {
57              if (objectRole.code.equals(codeString)) {
58                  return objectRole;
59              }
60          }
61  
62          throw new FHIRException("Unknown V3NullFlavor code '" + codeString + "'");
63      }
