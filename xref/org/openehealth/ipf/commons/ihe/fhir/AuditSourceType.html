10  public enum AuditSourceType {
11  
12      _1("1", "End-user display device, diagnostic device.", "User Device"),
13      _2("2", "Data acquisition device or instrument.", "Data Interface"),
14      _3("3", "Web Server process or thread.", "Web Server"),
15      _4("4", "Application Server process or thread.", "Application Server"),
16      _5("5", "Database Server process or thread.", "Database Server"),
17      _6("6", "Security server, e.g. a domain controller.", "Security Server"),
18      _7("7", "ISO level 1-3 network component.", "Network Device"),
19      _8("8", "ISO level 4-6 operating software.", "Network Router"),
20      _9("9", "other kind of device (defined by DICOM, but some other code/system can be used).", "Other"),
21      NULL(null, "?", "?");
22  
23      private String code;
24      private String definition;
25      private String display;
26  
27      AuditSourceType(String code, String definition, String display) {
28          this.code = code;
29          this.definition = definition;
30          this.display = display;
31      }
32  
33      public String getSystem() {
34          return "http://hl7.org/fhir/security-source-type";
35      }
36  
37      public static AuditSourceType fromCode(String codeString) throws FHIRException {
38          if (codeString == null || "".equals(codeString))
39              return null;
40  
41          for (AuditSourceType auditSourceType : AuditSourceType.values()) {
42              if (auditSourceType.code.equals(codeString)) {
43                  return auditSourceType;
44              }
45          }
46  
47          throw new FHIRException("Unknown AuditSourceType code '" + codeString + "'");
48      }
49  
50      public String toCode() {
51          return code;
52      }
53  
54  
55      public String getDefinition() {
56          return definition;
57  
58      }
59  
60      public String getDisplay() {
61          return display;
62      }
63  
64  
65  }
