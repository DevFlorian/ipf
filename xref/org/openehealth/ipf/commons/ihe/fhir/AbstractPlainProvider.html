32  /**
33   * Abstract plain provider that allows subclasses to forward the received payload into the
34   * Camel route served by the consumer. Note that this can be subclassed for writing so-called
35   * plain providers, while resource-specific providers should extend from {@link AbstractResourceProvider}.
36   *
37   * @author Christian Ohr
38   * @since 3.1
39   */
40  public abstract class AbstractPlainProvider implements Serializable {
41  
42      private static final Logger LOG = LoggerFactory.getLogger(AbstractPlainProvider.class);
43  
44      private RequestConsumer consumer;
45  
46      /**
47       *
48       * @param payload FHIR request resource
49       * @param resultType expected result type
50       * @param httpServletRequest servlet request
51       * @param httpServletResponse servlet response
52       * @param <R> Result type
53       * @return result of route processing
54       */
55      protected final <R extends IBaseResource> R requestResource(
56              Object payload, Class<R> resultType,
57              HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
58          return requestResource(payload, null, resultType, httpServletRequest, httpServletResponse);
59      }
60  
61      protected final <R extends IBaseResource> R requestResource(
62              Object payload, Map<String, Object> parameters, Class<R> resultType,
63              HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
64          if (consumer == null) {
65              throw new IllegalStateException("Consumer is not initialized");
66          }
67          Map<String, Object> headers = enrichParameters(parameters, httpServletRequest);
68          return consumer.handleResourceRequest(payload, headers, resultType);
69      }
70  
71      protected final <R extends IBaseResource> List<R> requestBundle(Object payload,
72              HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
73          return requestBundle(payload, null, httpServletRequest, httpServletResponse);
74      }
75  
76      protected final MethodOutcome requestAction(
77              Object payload,
78              HttpServletRequest httpServletRequest,
79              HttpServletResponse httpServletResponse)
80      {
81          return requestAction(payload, null, httpServletRequest, httpServletResponse);
82      }
83  
84      protected final MethodOutcome requestAction(
85              Object payload,
86              Map<String, Object> parameters,
87              HttpServletRequest httpServletRequest,
88              HttpServletResponse httpServletResponse)
89      {
90          if (consumer == null) {
91              throw new IllegalStateException("Consumer is not initialized");
92          }
93          Map<String, Object> headers = enrichParameters(parameters, httpServletRequest);
94          return consumer.handleAction(payload, headers);
95      }
96  
97      protected final Bundle requestTransaction(
98              Object payload,
99              Map<String, Object> parameters,
100             HttpServletRequest httpServletRequest,
101             HttpServletResponse httpServletResponse)
102     {
103         if (consumer == null) {
104             throw new IllegalStateException("Consumer is not initialized");
105         }
106         Map<String, Object> headers = enrichParameters(parameters, httpServletRequest);
107         return consumer.handleTransactionRequest(payload, headers);
108     }
109 
110     protected final <R extends IBaseResource> List<R> requestBundle(
111             Object payload, Map<String, Object> parameters,
112             HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
113         if (consumer == null) {
114             throw new IllegalStateException("Consumer is not initialized");
115         }
116         Map<String, Object> headers = enrichParameters(parameters, httpServletRequest);
117         return consumer.handleBundleRequest(payload, headers);
118     }
119 
120     private Map<String, Object> enrichParameters(Map<String, Object> parameters, HttpServletRequest httpServletRequest) {
121         // Populate some headers.
122         Map<String, Object> enriched = new HashMap<>();
123         enriched.put(HTTP_URI, httpServletRequest.getRequestURI());
124         enriched.put(HTTP_URL, httpServletRequest.getRequestURL());
125         enriched.put(HTTP_METHOD, httpServletRequest.getMethod());
126         enriched.put(HTTP_QUERY, httpServletRequest.getQueryString());
127         enriched.put(HTTP_CHARACTER_ENCODING, httpServletRequest.getCharacterEncoding());
128         enriched.put(HTTP_CONTENT_TYPE, httpServletRequest.getContentType());
129         enriched.put(HTTP_PROTOCOL_VERSION, httpServletRequest.getProtocol());
130         enriched.put(HTTP_CLIENT_IP_ADDRESS, httpServletRequest.getRemoteAddr());
131         enriched.put(HTTP_HEADERS, extractHttpHeaders(httpServletRequest));
132 
133         if (parameters == null) {
134             parameters = new HashMap<>();
135         }
136         enriched.put(FHIR_REQUEST_PARAMETERS, parameters);
137         return enriched;
138     }
139 
140     /**
141      * @param httpServletRequest
142      *      HTTP servlet request.
143      * @return
144      *      A map from header name to a list of header values.  This map is never <code>null</code>,
145      *      its values are never <code>null</code> and never empty.
146      */
147     private static Map<String, List<String>> extractHttpHeaders(HttpServletRequest httpServletRequest) {
148         Map<String, List<String>> result = new HashMap<>();
149         Enumeration<String> enumNames = httpServletRequest.getHeaderNames();
150         if (enumNames != null) {
151             while (enumNames.hasMoreElements()) {
152                 String name = enumNames.nextElement();
153                 Enumeration<String> enumValues = httpServletRequest.getHeaders(name);
154                 if (enumValues != null) {
155                     List<String> list = new ArrayList<>();
156                     while (enumValues.hasMoreElements()) {
157                         list.add(enumValues.nextElement());
158                     }
159                     if (! list.isEmpty()) {
160                         result.put(name, list);
161                     }
162                 }
163             }
164         }
165         return result;
166     }
167 
168     /**
169      * @return the configured consumer
170      */
171     public RequestConsumer getConsumer() {
172         return consumer;
173     }
174 
175     // Ensure this is only used once!
176     public void setConsumer(RequestConsumer consumer) {
177         if (this.consumer != null) {
178             throw new IllegalStateException("This provider is already used by a different consumer: " + consumer);
179         }
180         this.consumer = consumer;
181         LOG.info("Connected consumer {} to provider {}", consumer, this);
182     }
183 
184     public void unsetConsumer(RequestConsumer consumer) {
185         if (this.consumer == consumer) {
186             this.consumer = null;
187             LOG.info("Disconnected consumer {} from provider {}", consumer, this);
188         }
189     }
190 }
