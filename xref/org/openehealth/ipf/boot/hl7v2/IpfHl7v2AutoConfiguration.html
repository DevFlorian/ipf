60  /**
61   * Configure a basic IPF setup, mostly configuring HL7v2 and Mapping stuff
62   */
63  @Configuration
64  @AutoConfigureAfter(IpfAtnaAutoConfiguration.class)
65  @EnableConfigurationProperties(IpfHl7v2ConfigurationProperties.class)
66  public class IpfHl7v2AutoConfiguration {
67  
68      private static final String IPF_HL7_DEFINITIONS_PREFIX = "org.openehealth.ipf.commons.ihe.hl7v2.definitions";
69  
70      @Autowired
71      private IpfHl7v2ConfigurationProperties properties;
72  
73      @Bean
74      @ConditionalOnMissingBean(HL7MLLPCodec.class)
75      HL7MLLPCodec hl7codec(IpfHl7v2ConfigurationProperties config) {
76          HL7MLLPCodec hl7MLLPCodec = new CustomHL7MLLPCodec();
77          if (config.getCharset() != null) {
78              hl7MLLPCodec.setCharset(config.getCharset());
79          }
80          hl7MLLPCodec.setConvertLFtoCR(config.isConvertLinefeed());
81          return hl7MLLPCodec;
82      }
83  
84      @Bean
85      @ConditionalOnMissingBean(CustomModelClassFactory.class)
86      public CustomModelClassFactory mllpModelClassFactory() {
87          Map<String, String[]> eventMap = new HashMap<>();
88          eventMap.put("2.3.1", new String[] {
89                  IPF_HL7_DEFINITIONS_PREFIX + "pix.v231"
90          });
91          eventMap.put("2.5", new String[] {
92                  IPF_HL7_DEFINITIONS_PREFIX + "pdq.v25",
93                  IPF_HL7_DEFINITIONS_PREFIX + "pix.v25"
94          });
95          CustomModelClassFactory modelClassFactory = new CustomModelClassFactory(eventMap);
96          modelClassFactory.setEventMapDirectory("org/openehealth/ipf/commons/ihe/hl7v2/");
97          return modelClassFactory;
98      }
99  
100     @Bean
101     @ConditionalOnMissingBean(ProfileStore.class)
102     public ProfileStore profileStore() {
103         return new ClasspathProfileStore("/org/openehealth/ipf/gazelle/validation/profile/v2");
104     }
105 
106     @Bean
107     @ConditionalOnMissingBean(ValidationContext.class)
108     public ValidationContext validationContext() {
109         return ValidationContextFactory.noValidation();
110     }
111 
112     @Bean
113     @ConditionalOnMissingBean(IDGenerator.class)
114     @ConditionalOnProperty(prefix = "ipf.hl7v2", name = "generator", havingValue = "file", matchIfMissing = true)
115     public IDGenerator fileGenerator() {
116         return new IpfHiLoIdGenerator(properties.getIdGenerator());
117     }
118 
119     @Bean
120     @ConditionalOnMissingBean(IDGenerator.class)
121     @ConditionalOnProperty(prefix = "ipf.hl7v2", name = "generator", havingValue = "uuid")
122     public IDGenerator uuidGenerator() {
123         return new UUIDGenerator();
124     }
125 
126     @Bean
127     @ConditionalOnMissingBean(IDGenerator.class)
128     @ConditionalOnProperty(prefix = "ipf.hl7v2", name = "generator", havingValue = "nano")
129     public IDGenerator nanoGenerator() {
130         return new NanoTimeGenerator();
131     }
132 
133     @ConfigurationProperties(prefix = "ipf.hl7v2.parser")
134     @Bean
135     public ParserConfiguration parserConfiguration() {
136         return new ParserConfiguration();
137     }
138 
139     @Bean
140     @ConditionalOnMissingBean(HapiContext.class)
141     public HapiContext hapiContext(CustomModelClassFactory modelClassFactory, ProfileStore profileStore,
142                                    ValidationContext validationContext, ParserConfiguration parserConfiguration,
143                                    IDGenerator idGenerator) {
144         HapiContext context = new DefaultHapiContext();
145         context.setModelClassFactory(modelClassFactory);
146         context.setValidationContext(validationContext);
147         context.setProfileStore(profileStore);
148         context.setParserConfiguration(parserConfiguration);
149         context.getParserConfiguration().setIdGenerator(idGenerator);
150         context.getParserConfiguration().setEscaping(DefaultEscaping.INSTANCE);
151         return context;
152     }
153 
154     // Provide bean for MLLP endpoint dispatching
155     @Bean
156     @ConditionalOnMissingBean(ConsumerDispatchingInterceptor.class)
157     public ConsumerDispatchingInterceptor mllpDispatcher(CamelContext camelContext) {
158         return new ConsumerDispatchingInterceptor(camelContext);
159     }
160 
161     // Provide "interactiveContinuationStorage" for HL7v2 paging
162     @Bean
163     @ConditionalOnMissingBean(InteractiveContinuationStorage.class)
164     @ConditionalOnSingleCandidate(CacheManager.class)
165     @ConditionalOnProperty("ipf.hl7v2.caching")
166     public InteractiveContinuationStorage interactiveContinuationStorage(CacheManager cacheManager) {
167         return new CachingInteractiveHl7v2ContinuationStorage(cacheManager);
168     }
169 
170     // Provide "unsolicitedFragmentationStorage" for HL7v2 fragmentation
171     @Bean
172     @ConditionalOnMissingBean(UnsolicitedFragmentationStorage.class)
173     @ConditionalOnSingleCandidate(CacheManager.class)
174     @ConditionalOnProperty("ipf.hl7v2.caching")
175     public UnsolicitedFragmentationStorage unsolicitedFragmentationStorage(CacheManager cacheManager) {
176         return new CachingUnsolicitedFragmentionStorage(cacheManager);
177     }
