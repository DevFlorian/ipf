31  
32  import javax.xml.namespace.QName;
33  import java.util.Map;
34  
35  /**
36   * The Camel component for the ITI-55 transaction (XCPD).
37   */
38  public class Iti55Component extends Hl7v3Component<Hl7v3WsTransactionConfiguration> {
39      private final static String NS_URI = "urn:ihe:iti:xcpd:2009";
40      public final static Hl7v3WsTransactionConfiguration WS_CONFIG = new Hl7v3WsTransactionConfiguration(
41              IpfInteractionId.ITI_55,
42              new QName(NS_URI, "RespondingGateway_Service", "xcpd"),
43              Iti55PortType.class,
44              new QName(NS_URI, "RespondingGateway_Binding_Soap12", "xcpd"),
45              false,
46              "wsdl/iti55/iti55-raw.wsdl",
47              "PRPA_IN201306UV02",
48              "PRPA_TE201306UV02",
49              false,
50              true);
51  
52  
53      @SuppressWarnings({"raw", "unchecked"}) // Required because of base class
54      @Override
55      protected Endpoint createEndpoint(String uri, String remaining, Map parameters) throws Exception {
56          return new Hl7v3Endpoint<Hl7v3WsTransactionConfiguration>(uri, remaining, this,
57                  getCustomInterceptors(parameters),
58                  getFeatures(parameters),
59                  getSchemaLocations(parameters),
60                  getProperties(parameters),
61                  null) {
62              @Override
63              public AbstractWsProducer<Hl7v3AuditDataset, Hl7v3WsTransactionConfiguration, ?, ?> getProducer(AbstractWsEndpoint<Hl7v3AuditDataset, Hl7v3WsTransactionConfiguration> endpoint,
64                                                    JaxWsClientFactory<Hl7v3AuditDataset> clientFactory) {
65                  return new Iti55Producer(endpoint, clientFactory);
66              }
67  
68              @Override
69              protected AbstractWebService getCustomServiceInstance(AbstractWsEndpoint<Hl7v3AuditDataset, Hl7v3WsTransactionConfiguration> endpoint) {
70                  return new Iti55Service((Hl7v3Endpoint<Hl7v3WsTransactionConfiguration>) endpoint);
71              }
72          };
73      }
74  
75      @Override
76      public Hl7v3WsTransactionConfiguration getWsTransactionConfiguration() {
77          return WS_CONFIG;
78      }
79  
80      @Override
81      public AuditStrategy<Hl7v3AuditDataset> getClientAuditStrategy() {
82          return new Iti55AuditStrategy(false);
83      }
84  
85      @Override
86      public AuditStrategy<Hl7v3AuditDataset> getServerAuditStrategy() {
87          return new Iti55AuditStrategy(true);
88      }
89  
90  }
