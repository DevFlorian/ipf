41  /**
42   * Camel endpoint for HL7v2-WS transaction with a single operation.
43   */
44  public class SimpleHl7v2WsEndpoint extends AbstractWsEndpoint<AbstractHl7v2WsComponent> {
45  
46      /**
47       * Constructs the endpoint.
48       * @param endpointUri
49       *          the endpoint URI.
50       * @param address
51       *          the endpoint address from the URI.
52       * @param component
53       *          the component creating this endpoint.
54       * @param customInterceptors
55       *          user-defined CXF interceptors.
56       * @param features
57       *          user-defined list of CXF features.
58       */
59      public SimpleHl7v2WsEndpoint(
60              String endpointUri,
61              String address,
62              AbstractHl7v2WsComponent component,
63              InterceptorProvider customInterceptors,
64              List<AbstractFeature> features,
65              List<String> schemaLocations,
66              Map<String, Object> properties)
67      {
68          super(endpointUri, address, component, customInterceptors, features, schemaLocations, properties);
69      }
70  
71  
72      protected List<Hl7v2Interceptor> getProducerInterceptorChain() {
73          return Arrays.<Hl7v2Interceptor> asList(
74                  new ProducerMarshalInterceptor(),
75                  new ProducerResponseAcceptanceInterceptor(),
76                  new ProducerRequestAcceptanceInterceptor(),
77                  new ProducerAdaptingInterceptor()
78          );
79      }
80  
81  
82      @Override
83      public Producer createProducer() throws Exception {
84          return Hl7v2InterceptorUtils.adaptProducerChain(
85                  getProducerInterceptorChain(),
86                  getComponent(),
87                  getComponent().getProducer(this, getJaxWsClientFactory()));
88      }
89  
90  
91      @Override
92      public Consumer createConsumer(Processor processor) throws Exception {
93          AbstractHl7v2WebService serviceInstance =
94                  (AbstractHl7v2WebService) getComponent().getServiceInstance(this);
95          serviceInstance.setHl7v2Configuration(getComponent());
96          ServerFactoryBean serverFactory = getJaxWsServiceFactory().createServerFactory(serviceInstance);
97          Server server = serverFactory.create();
98          AbstractWebService service = (AbstractWebService) serverFactory.getServiceBean();
99          return new DefaultWsConsumer(this, processor, service, server);
100     }
101 
102 
103     @Override
104     public JaxWsClientFactory getJaxWsClientFactory() {
105         return new JaxWsClientFactory(
106                 getComponent().getWsTransactionConfiguration(),
107                 getServiceUrl(),
108                 null,
109                 getCustomInterceptors());
110     }
111 
112 
113     @Override
114     public JaxWsServiceFactory getJaxWsServiceFactory() {
115         return new JaxWsServiceFactory(
116                 getComponent().getWsTransactionConfiguration(),
117                 getServiceAddress(),
118                 null,
119                 getCustomInterceptors(),
120                 getRejectionHandlingStrategy());
121     }
122 }
