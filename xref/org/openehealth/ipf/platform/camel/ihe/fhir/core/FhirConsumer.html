30  import org.openehealth.ipf.commons.ihe.fhir.FhirAuditDataset;
31  import org.openehealth.ipf.commons.ihe.fhir.RequestConsumer;
32  import org.openehealth.ipf.platform.camel.core.util.Exchanges;

37  /**
38   * FHIR consumer, which is an implementation of a {@link RequestConsumer} that handles requests
39   * by sending the request data and parameters into a Camel route and returning the result of
40   * the route processing.
41   *
42   * @author Christian Ohr
43   * @since 3.1
44   */
45  public class FhirConsumer<AuditDatasetType extends FhirAuditDataset> extends DefaultConsumer
46          implements SuspendableService, RequestConsumer {
47  
48      public FhirConsumer(FhirEndpoint<AuditDatasetType, ? extends FhirComponent<AuditDatasetType>> endpoint, Processor processor) {
49          super(endpoint, processor);
50      }
51  
52      @Override
53      protected void doStart() throws Exception {
54          super.doStart();
55          this.getEndpoint().connect(this);
56      }
57  
58      @Override
59      protected void doStop() throws Exception {
60          this.getEndpoint().disconnect(this);
61          super.doStop();
62      }
63  
64      @Override
65      public FhirEndpoint<AuditDatasetType, FhirComponent<AuditDatasetType>> getEndpoint() {
66          return (FhirEndpoint<AuditDatasetType, FhirComponent<AuditDatasetType>>) super.getEndpoint();
67      }
68  
69      /**
70       * This method can be called by {@link ca.uhn.fhir.rest.server.IResourceProvider} objects to send the received
71       * (and potentially handled) request further down a Camel route.
72       *
73       * @param payload     FHIR request content
74       * @param headers     headers
75       * @param resultClass class of the result resource
76       * @param <R>         Resource type being returned
77       * @return result of processing the FHIR request in Camel
78       */
79      @Override
80      public final <R extends IBaseResource> R handleResourceRequest(Object payload, Map<String, Object> headers, Class<R> resultClass) {
81          return handleInRoute(payload, headers, resultClass);
82      }
83  
84      @Override
85      public <R extends IBaseResource> List<R> handleBundleRequest(Object payload, Map<String, Object> headers) {
86          return (List<R>) handleInRoute(payload, headers, List.class);
87      }
88  
89      @Override
90      public Bundle handleTransactionRequest(Object payload, Map<String, Object> headers) {
91          return handleInRoute(payload, headers, Bundle.class);
92      }
93  
94      @Override
95      public MethodOutcome handleAction(Object payload, Map<String, Object> headers) {
96          return handleInRoute(payload, headers, MethodOutcome.class);
97      }
98  
99      /**
100      * Forwards the request to be handled into a Camel route
101      *
102      * @param payload     request payload, will become the Camel message body
103      * @param headers     request parameters, will be added to the Camel headers
104      * @param resultClass expected body type to be returned
105      * @return request result, type-converted into the required result class
106      */
107     protected <T> T handleInRoute(Object payload, Map<String, Object> headers, Class<T> resultClass) {
108         Exchange exchange = getEndpoint().createExchange();
109         exchange.getIn().setBody(payload);
110         if (headers != null) {
111             exchange.getIn().getHeaders().putAll(headers);
112         }
113 
114         // Add the FHIR context as header
115         exchange.getIn().getHeaders().put(Constants.FHIR_CONTEXT, getEndpoint().getInterceptableConfiguration().getContext());
116 
117         try {
118             getProcessor().process(exchange);
119         } catch (Throwable e) {
120             getExceptionHandler().handleException(e);
121         }
122 
123         // If the exchange has failed, throw the exception back into the servlet
124         if (exchange.isFailed()) {
125             BaseServerResponseException e = exchange.getException(BaseServerResponseException.class);
126             throw (e != null) ? e : new InternalErrorException("Unexpected server error", exchange.getException());
127         }
128 
129         Message resultMessage = Exchanges.resultMessage(exchange);
130         return getEndpoint().getCamelContext().getTypeConverter()
131                 .convertTo(resultClass, exchange, resultMessage.getBody());
132     }
133 }
