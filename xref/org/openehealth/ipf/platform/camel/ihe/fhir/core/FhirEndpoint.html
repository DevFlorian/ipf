17  package org.openehealth.ipf.platform.camel.ihe.fhir.core;
18  
19  import ca.uhn.fhir.context.FhirContext;
20  import org.apache.camel.Consumer;
21  import org.apache.camel.ExchangePattern;
22  import org.apache.camel.Processor;
23  import org.apache.camel.Producer;
24  import org.apache.camel.api.management.ManagedAttribute;
25  import org.openehealth.ipf.commons.ihe.core.atna.AuditStrategy;
26  import org.openehealth.ipf.commons.ihe.fhir.AbstractPlainProvider;
27  import org.openehealth.ipf.commons.ihe.fhir.CamelFhirServlet;
28  import org.openehealth.ipf.commons.ihe.fhir.ClientRequestFactory;
29  import org.openehealth.ipf.commons.ihe.fhir.FhirAuditDataset;
30  import org.openehealth.ipf.platform.camel.ihe.atna.AuditableEndpoint;
31  import org.openehealth.ipf.platform.camel.ihe.core.InterceptableEndpoint;
32  import org.openehealth.ipf.platform.camel.ihe.core.Interceptor;
33  import org.openehealth.ipf.platform.camel.ihe.fhir.core.intercept.consumer.ConsumerAuditInterceptor;
34  import org.openehealth.ipf.platform.camel.ihe.fhir.core.intercept.producer.ProducerAuditInterceptor;
35  
36  import java.util.ArrayList;
37  import java.util.List;
38  
39  /**
40   * Generic FHIR endpoint
41   *
42   * @author Christian Ohr
43   * @since 3.1
44   */
45  public abstract class FhirEndpoint<AuditDatasetType extends FhirAuditDataset, ComponentType extends FhirComponent<AuditDatasetType>>
46          extends InterceptableEndpoint<FhirEndpointConfiguration<AuditDatasetType>, ComponentType>
47          implements AuditableEndpoint<AuditDatasetType> {
48  
49      private final FhirEndpointConfiguration<AuditDatasetType> config;
50      private String servletName;
51      private final ComponentType fhirComponent;
52  
53      public FhirEndpoint(String uri, ComponentType fhirComponent, FhirEndpointConfiguration<AuditDatasetType> config) {
54          super(uri, fhirComponent);
55          this.fhirComponent = fhirComponent;
56          this.config = config;
57          this.servletName = config.getServletName();
58          this.setExchangePattern(ExchangePattern.InOut);
59      }
60  
61      @Override
62      protected ComponentType getInterceptableComponent() {
63          return fhirComponent;
64      }
65  
66      @Override
67      protected Producer doCreateProducer() throws Exception {
68          return new FhirProducer<AuditDatasetType>(this);
69      }
70  
71      /**
72       * Called when a {@link FhirConsumer} is started. Registers the resource provider
73       *
74       * @param consumer FhirConsumer
75       * @throws Exception
76       */
77      public void connect(FhirConsumer<AuditDatasetType> consumer) throws Exception {
78          AbstractPlainProvider provider = getResourceProvider();
79          // Make consumer known to provider
80          provider.setConsumer(consumer);
81          // Register provider with CamelFhirServlet
82          CamelFhirServlet.registerProvider(servletName, provider);
83      }
84  
85      /**
86       * Called when a {@link FhirConsumer} is stopped. Unregisters the resource provider
87       *
88       * @param consumer FhirConsumer
89       * @throws Exception
90       */
91      public void disconnect(FhirConsumer<AuditDatasetType> consumer) throws Exception {
92          AbstractPlainProvider provider = getResourceProvider();
93          CamelFhirServlet.unregisterProvider(servletName, provider);
94          provider.unsetConsumer(consumer);
95      }
96  
97      public FhirContext getContext() {
98          return getInterceptableConfiguration().getContext();
99      }
100 
101     /**
102      * Returns a list of interceptors that are default for FHIR consumers. Subclasses
103      * can add additional interceptors that are required for a concrete FHIR endpoint.
104      *
105      * @return list of default interceptors
106      */
107     @Override
108     protected List<Interceptor> createInitialConsumerInterceptorChain() {
109         List<Interceptor> initialChain = new ArrayList<>();
110         if (isAudit()) {
111             initialChain.add(new ConsumerAuditInterceptor<>());
112         }
113         return initialChain;
114     }
115 
116     /**
117      * Returns a list of interceptors that are default for FHIR producers. Subclasses
118      * can add additional interceptors that are required for a concrete FHIR endpoint.
119      *
120      * @return list of default interceptors
121      */
122     @Override
123     protected List<Interceptor> createInitialProducerInterceptorChain() {
124         List<Interceptor> initialChain = new ArrayList<>();
125         if (isAudit()) {
126             initialChain.add(new ProducerAuditInterceptor<>());
127         }
128         return initialChain;
129     }
130 
131     /**
132      * Returns <tt>true</tt> when ATNA auditing should be performed.
133      */
134     @Override
135     @ManagedAttribute(description = "Audit Enabled")
136     public boolean isAudit() {
137         return config.isAudit();
138     }
139 
140     @Override
141     public boolean isSingleton() {
142         return true;
143     }
144 
145     public FhirComponentConfiguration getFhirComponentConfiguration() {
146         return fhirComponent.getFhirComponentConfiguration();
147     }
148 
149     @Override
150     protected FhirEndpointConfiguration<AuditDatasetType> getInterceptableConfiguration() {
151         return config;
152     }
153 
154     /**
155      * Returns client-side audit strategy instance.
156      */
157     @Override
158     public AuditStrategy<AuditDatasetType> getClientAuditStrategy() {
159         return fhirComponent.getClientAuditStrategy();
160     }
161 
162     /**
163      * Returns server-side audit strategy instance.
164      */
165     @Override
166     public AuditStrategy<AuditDatasetType> getServerAuditStrategy() {
167         return fhirComponent.getServerAuditStrategy();
168     }
169 
170     @Override
171     public Consumer doCreateConsumer(Processor processor) throws Exception {
172         return new FhirConsumer<>(this, processor);
173     }
174 
175     // Private stuff
176 
177     private AbstractPlainProvider getResourceProvider() {
178         AbstractPlainProvider provider = config.getResourceProvider();
179         if (provider == null) {
180             provider = getFhirComponentConfiguration().getStaticResourceProvider();
181         }
182         return provider;
183     }
184 
185     public ClientRequestFactory<?> getClientRequestFactory() {
186         ClientRequestFactory<?> factory = config.getClientRequestFactory();
187         if (factory == null) {
188             factory = getFhirComponentConfiguration().getStaticClientRequestFactory();
189         }
190         return factory;
191     }
192 
193 }
